\begin{frame}[fragile]

pen
#=> "A trusty purple pen"
\end{rubycode}
\end{frame}

\begin{frame}[fragile]

    "<p>#{thing}</p>"
  end
end
\end{rubycode}

Scribe = Class.new do

\begin{frame}[fragile]
